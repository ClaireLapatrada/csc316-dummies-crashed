2.3 WRANGLEDATA METHOD
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
PARAMETERS: None

PURPOSE: Process all data and prepare counts by type for all years

LOGIC:
  // Initialize data structure to hold counts for all years
  vis.yearlyData = {}

  console.log('SignViz: Processing all data...')

  // Group data by year and count crash types
  vis.data.forEach(d => {
    const year = d.year || d.Year

    if (!vis.yearlyData[year]) {
      vis.yearlyData[year] = {
        automobile: 0,
        cyclist: 0,
        pedestrian: 0
      }
    }

    // Check each involvement type
    // Automobile ONLY (not involving cyclist or pedestrian)
    const autoOnly = (d['Automobile Involved'] === 'Yes' || d['Automobile Involved'] === '1')
                   && d['Cyclist Involved'] !== 'Yes'
                   && d['Pedestrian Involved'] !== 'Yes'

    const cyclistInvolved = d['Cyclist Involved'] === 'Yes' || d['Cyclist Involved'] === '1'
    const pedestrianInvolved = d['Pedestrian Involved'] === 'Yes' || d['Pedestrian Involved'] === '1'

    if (autoOnly) vis.yearlyData[year].automobile++
    if (cyclistInvolved) vis.yearlyData[year].cyclist++
    if (pedestrianInvolved) vis.yearlyData[year].pedestrian++
  })

  console.log('SignViz: Yearly data processed:', vis.yearlyData)

  // Update with current year
  vis.updateVis(vis.currentYear)

LOGIC:
  console.log(`SignViz: Updating to year ${year}`)

  console.log('SignViz counts for', year, ':', counts)

  // Create array and sort by count (descending)
  vis.displayData = [
    { type: 'automobile', count: counts.automobile, ...vis.signs[0] },
    { type: 'cyclist', count: counts.cyclist, ...vis.signs[1] },
    { type: 'pedestrian', count: counts.pedestrian, ...vis.signs[2] }
  ].sort((a, b) => b.count - a.count)

  // Assign rankings
  vis.displayData.forEach((d, i) => {
    d.rank = i + 1
  })

  console.log('SignViz ranked data:', vis.displayData)

TRANSITION SETUP:
  const t = d3.transition()
    .duration(1000)
    .ease(d3.easeCubicInOut)

    })
